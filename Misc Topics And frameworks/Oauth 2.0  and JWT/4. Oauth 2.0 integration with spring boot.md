```java
package com.example.googlephotosfetcher.config; // Package for your configuration classes

import org.springframework.context.annotation.Bean; // Annotation to create beans
import org.springframework.context.annotation.Configuration; // Marks this as a configuration class
import org.springframework.security.config.annotation.web.builders.HttpSecurity; // Configure web security
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity; // Enable Spring Security
import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserService; // OIDC user service for fetching info
import org.springframework.security.web.SecurityFilterChain; // Represents a chain of security filters

@Configuration // This is a Spring configuration class
@EnableWebSecurity // Enables Spring Security's web security support
public class SecurityConfig {

    @Bean // Declare a bean named "securityFilterChain"
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception { 
        // Configure how HTTP requests should be handled

        http
            .authorizeRequests(authorizeRequests ->  // Start configuring authorization rules
                authorizeRequests
                    .antMatchers("/", "/login**", "/error**").permitAll() // Allow anyone to access these URLs
                    .anyRequest().authenticated()  // All other requests require authentication
            )
            .oauth2Login(oauth2Login ->   // Configure the OAuth 2.0 login process
                oauth2Login.userInfoEndpoint(userInfoEndpoint -> 
                    userInfoEndpoint.oidcUserService(new OidcUserService()) // Fetch user info with OIDC after login
                )
            );

        return http.build(); // Build and return the security filter chain
    }
}
```
