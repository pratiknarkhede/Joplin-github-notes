import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.*;
import java.util.concurrent.atomic.AtomicInteger;

public class ExecutorServiceDemo {
    
    // A counter to track completed tasks
    private static final AtomicInteger completedTasks = new AtomicInteger(0);
    
    // Simulates a task that takes some time and returns a result
    private static class ComputeTask implements Callable<Integer> {
        private final int taskId;
        private final int complexity;
        
        public ComputeTask(int taskId, int complexity) {
            this.taskId = taskId;
            this.complexity = complexity;
        }
        
        @Override
        public Integer call() throws Exception {
            System.out.println("Task " + taskId + " started on thread: " + 
                              Thread.currentThread().getName());
            
            // Simulate CPU-intensive work
            int result = 0;
            for (int i = 0; i < complexity * 10_000_000; i++) {
                result += (i % 10);
            }
            
            // Track completion
            completedTasks.incrementAndGet();
            
            System.out.println("Task " + taskId + " completed on thread: " + 
                              Thread.currentThread().getName());
            
            return taskId * 100 + (result % 100); // Return a value based on task ID
        }
    }
    
    public static void main(String[] args) {
        int processors = Runtime.getRuntime().availableProcessors();
        System.out.println("Running on a machine with " + processors + " processors/cores");
        
        // Create a thread pool with the optimal size (typically number of cores)
        ExecutorService executor = Executors.newFixedThreadPool(processors);
        
        try {
            // Create a list of tasks with varying complexity
            List<Callable<Integer>> tasks = new ArrayList<>();
            for (int i = 1; i <= 10; i++) {
                tasks.add(new ComputeTask(i, i % 3 + 1)); // Complexity 1-3
            }
            
            System.out.println("Submitting 10 tasks to the executor...");
            
            // Submit all tasks and get Future objects
            long startTime = System.currentTimeMillis();
            List<Future<Integer>> futures = executor.invokeAll(tasks);
            
            // Process the results as they complete
            int sum = 0;
            for (int i = 0; i < futures.size(); i++) {
                // This will block until the specific task completes
                Integer result = futures.get(i).get();
                System.out.println("Result from task " + (i+1) + ": " + result);
                sum += result;
            }
            
            long endTime = System.currentTimeMillis();
            System.out.println("\nAll tasks completed!");
            System.out.println("Total execution time: " + (endTime - startTime) + "ms");
            System.out.println("Sum of all results: " + sum);
            
            // Demonstrate the ScheduledExecutorService
            demonstrateScheduling();
            
        } catch (InterruptedException | ExecutionException e) {
            System.err.println("Error executing tasks:");
            e.printStackTrace();
        } finally {
            // Always shut down the executor
            System.out.println("\nShutting down executor...");
            shutdownAndAwaitTermination(executor);
        }
    }
    
    private static void demonstrateScheduling() {
        System.out.println("\n--- Scheduling Demo ---");
        ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(1);
        
        // Run a task after a delay
        System.out.println("Scheduling a task to run after 2 seconds...");
        ScheduledFuture<?> future = scheduler.schedule(
            () -> System.out.println("Delayed task executed at: " + System.currentTimeMillis()),
            2, 
            TimeUnit.SECONDS
        );
        
        try {
            // Wait for the task to complete
            future.get();
            
            // Schedule a periodic task
            System.out.println("Starting a task that runs every second...");
            scheduler.scheduleAtFixedRate(() -> {
                System.out.println("Periodic task executed at: " + System.currentTimeMillis());
            }, 0, 1, TimeUnit.SECONDS);
            
            // Let the periodic task run for a few seconds
            Thread.sleep(5000);
            
        } catch (InterruptedException | ExecutionException e) {
            e.printStackTrace();
        } finally {
            System.out.println("Shutting down scheduler...");
            shutdownAndAwaitTermination(scheduler);
        }
    }
    
    // Utility method to shut down an ExecutorService
    private static void shutdownAndAwaitTermination(ExecutorService executor) {
        // Disable new tasks from being submitted
        executor.shutdown();
        try {
            // Wait for existing tasks to terminate
            if (!executor.awaitTermination(10, TimeUnit.SECONDS)) {
                // Cancel currently executing tasks forcefully
                executor.shutdownNow();
                // Wait a while for tasks to respond to being cancelled
                if (!executor.awaitTermination(10, TimeUnit.SECONDS)) {
                    System.err.println("Executor did not terminate");
                }
            }
        } catch (InterruptedException e) {
            // (Re-)Cancel if current thread also interrupted
            executor.shutdownNow();
            // Preserve interrupt status
            Thread.currentThread().interrupt();
        }
    }
}
